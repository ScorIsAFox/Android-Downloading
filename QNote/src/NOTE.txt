int old_x;
			int old_y;
			public void mouseDragged(MouseEvent e) {
				IJFrame.this.setLocation(IJFrame.this.getLocation().x+e.getX()-old_x, 
						IJFrame.this.getLocation().y+e.getY()-old_y);
			}
 
			public void mouseMoved(MouseEvent e) {
				old_x = e.getX();
				old_y = e.getY();
			}
